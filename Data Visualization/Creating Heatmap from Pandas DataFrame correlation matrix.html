import pandas as pd
import seaborn as sns
%matplotlib inline

data = {
    "x": range(0, 100),
    "y": range(100, 0, -1),
    "z": range(0, 100),
    "a": [4, 5] * 50,
    "b": [4, 5, 6, 7] * 25
}

df = pd.DataFrame(data)
print(df.head())

corr = df.corr()
print(corr)
# plot the heatmap
sns.heatmap(corr, xticklabels=corr.columns, yticklabels=corr.columns)
